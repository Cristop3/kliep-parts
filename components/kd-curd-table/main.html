  MAPPING_ROW_GUTTER,

interface IKdCurdTableProps {
  newDisabled?: boolean; // “新增”disabled
  editDisabled?: boolean; // “编辑”disabled
  deleteDisabled?: boolean; // “删除”disabled

  editShow?: boolean; // 是否显示“编辑”
  deleteShow?: boolean; // 是否显示“删除”
  newShow?: boolean; // 是否显示“新增”
  noBtn?: boolean; // 是否所有操作按钮不显示

  formFields: any; // 表单初始化值
  formRules?: Function; // 透传表单规则回调构建 主要为了透传formModel出去

  dialogTitle?: string; // 弹框名称
}
