kd-upload配置项 ​

typescript
export interface IKdUploadConfig {
  downloadUrl: string; // 下载url

  splitFlag?: string; // 多个附件分割标识 默认逗号“,”
  splicingFlag?: string; // 文件name | url | ... 等等配置参数分割标识

  fileParams: string; // 1. 强制第0位为文件名字段key,第1位为文件url路径字段key 2. 取接口哪些参数来组成当前文件对象和最终提交值的字符串拼接 如'name,url,type'对应v-model值为name=xxx||url=xxx||type=xxx 对应文件对象为{name:'xxx',url:'xxx',type:'xxx'}
  responseMapping: { success: string; data: string; errorMsg: string }; // 需配置接口1. 成功标识字段 2. 文件内容字段 3. 错误信息字段
}
