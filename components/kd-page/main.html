modal页面 ​

modal页面是指可指定当前假页面挂载到任何指定的dom下面，正常的dialog太小了，不方便业务操作； ​

使用时必须传递teleportKey同时被指定的dom需设置position:relative；同时初始化时设置visible为false ​

<template>
  <el-button type="warning" size="small" @click="visible = true">
    打开modal页面
  </el-button>
  <kd-page
    name="modal页面"
    type="modal-page"
    teleport-key="#app"
    :visible="visible"
    @modal-cancel="visible = !visible">
    我是modal页面内容
  </kd-page>
</template>

<script setup lang="ts">
// ---------------------------------------- Import Business --------------------------------------------- //
import { ref } from "vue";

Page属性 ​

interface IPageProps {
  // common props
  name?: string; // 名称

  type?: "table-page" | "child-page" | "modal-page"; // 页面类型table-page挂载表格；child-page子页面；modal-page模态框页面

  usePadding?: boolean; // 是否沿用系统业务样式padding table使用 详情页设置为false

  footerBtn?: boolean; // 是否存在页面按钮 一般用于子页面详情 table-page一般不需要 详情页设置为true

  useScroll?: boolean; // 是否对content设置overflow-y:auto 主要是针对modal-page组件只在内容区滚动

  loading?: boolean; // loading 特指内容区域loading

  // modal props
  teleportKey?: string; // 针对modal-page时需要设置将modal挂载到哪里 最好保持透传的dom是唯一 目前只支持传递class 并且保证所传递的dom自身有宽高
  visible?: boolean; // 显示modal-page
  modalConfirmLabel?: string; // 确认按钮label
  modalCancelLabel?: string; // 取消按钮label
  modalConfirmBtnShow?: boolean; // 是否显示确认按钮
  modalCancelBtnShow?: boolean; // 是否显示取消按钮
  modalConfirmBtnLoading?: boolean; // 确认按钮loading
}
