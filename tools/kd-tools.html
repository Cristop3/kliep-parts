/**
 * @function 统一构造filterList中的filterItem
 * @example 以下方法只是常用模式的提取，其他属性走{...input(...),xxx:''}
 * @back input // 统一构造input类型
 *    @param formItemLabel:string
 *    @param formItemKey:string
 * @back select // 统一构造select类型
 *    @param formItemLabel:string
 *    @param formItemKey:string
 *    @param selectOptions:any[] 下拉集合
 *    @param labelValue:{ label: string; value: string; join?: boolean } & null & undefined 针对需要改变下拉显示名称和值字段
 * @back datePicker // 统一构造datePicker类型
 *    @param formItemLabel:string
 *    @param formItemKey:string
 *    @param type:string // 默认走daterange 内部处理date|dates|datetimerange|daterange的formatter 其余类型需额外处理 支持全type
 */
const { input, select, datePicker } = useFilterList();
const filterList = ref([
  input("我是hooks-input", "test"),
  select("我是hooks-select", "test2", [{ label: "1", value: 1 }]),
  select(
    "我是hooks-select2",
    "test3",
    [
      { name: "1", code: 1 },
      { name: "2", code: 2 },
    ],
    { label: "name", value: "code" }
  ),
  datePicker("我是hooks-datePicker-daterange", "test4,test5"),
  datePicker("我是hooks-datePicker-date", "test6", "date"),
  datePicker(
    "我是hooks-datePicker-datetimerange",
    "test7,test8",
    "datetimerange"
  ),
